1use anyhow::Result;

8#[derive(Parser, Debug)]
9pub struct ConfigArgs {
10    #[command(subcommand)]
11    command: ConfigCommands,
12}

14#[derive(Subcommand, Debug)]
15enum ConfigCommands {
16    #[command(about = "Show current configuration")]
17    Show,
18
19    #[command(about = "Initialize a new configuration file")]
20    Init {
21        #[arg(
22            long,
23            help = "Output path for config file",
24            default_value = "paddi.toml"
25        )]
26        output: PathBuf,
27
28        #[arg(long, help = "Overwrite existing file")]
29        force: bool,
30    },
31
32    #[command(about = "Validate configuration")]
33    Validate {
34        #[arg(long, help = "Path to config file")]
35        file: Option<PathBuf>,
36    },
37}

39pub async fn run(args: ConfigArgs, config: Config) -> Result<()> {
40    match args.command {
41        ConfigCommands::Show => show_config(config).await,
42        ConfigCommands::Init { output, force } => init_config(output, force).await,
43        ConfigCommands::Validate { file } => validate_config(file, config).await,
44    }
45}

57async fn init_config(output: PathBuf, force: bool) -> Result<()> {
58    info!("Initializing configuration file");
59
60    // Check if file exists
61    if tokio::fs::try_exists(&output).await.unwrap_or(false) && !force {
62        anyhow::bail!(
63            "Configuration file already exists: {}. Use --force to overwrite.",
64            output.display()
65        );
66    }
67
68    // Create default config
69    let config = Config::default();
70
71    // Save to file
72    config.save(&output)?;
73
74    println!("‚úÖ Configuration file created: {}", output.display());
75    println!("\nExample configuration:");
76    println!("{}", toml::to_string_pretty(&config)?);
77
78    Ok(())
79}
