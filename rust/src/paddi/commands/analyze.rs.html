20pub async fn run(args: AnalyzeArgs, config: Config) -> Result<()> {
21    info!("Running explainer agent");
22
23    // Validation checks
24    if !args.skip_validation {
25        check_python_available(&config.python.command).await?;
26        check_agents_exist(&config.python.agents_path).await?;
27    }
28
29    // Check if input data exists
30    let input_file = config.paths.data_dir.join("collected.json");
31    if !tokio::fs::try_exists(&input_file).await.unwrap_or(false) {
32        anyhow::bail!(
33            "Input file not found: {}. Please run 'paddi collect' first.",
34            input_file.display()
35        );
36    }
37
38    // Create orchestrator
39    let orchestrator = AgentOrchestrator::new(config).with_progress();
40
41    // Run explainer
42    let result = orchestrator
43        .run_explainer(args.use_mock, args.project_id)
44        .await?;
45
46    if result.success {
47        info!("Analysis completed successfully");
48        println!("\nâœ… Analysis completed successfully!");
49        println!("ðŸ“„ Results saved to 'data/explained.json'");
50    } else {
51        anyhow::bail!("Analysis failed: {}", result.error);
52    }
53
54    Ok(())
